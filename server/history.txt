* mqtt - mosquito
docker run -it -d --name mos -p 1883:1883 \
eclipse-mosquitto:2
docker cp mosquitto.conf mos1:/mosquitto/config/mosquitto.conf

docker exec -it mos1 sh
# create pwd file, user
mosquitto_passwd -c /mosquitto/passwd_file user_name
# add user
mosquitto_passwd -b /mosquitto/passwd_file user_name password
# del user
mosquitto_passwd -D /mosquitto/passwd_file user_name

cd /mosquitto/config/
vi mosquitto.conf

allow_anonymous false
listener 1883 0.0.0.0

docker restart mos1
docker logs -f mos1
